/**
 * Hämtar statistik från ett externt JSON-bibliotek, filtrerar fram de 6 mest sökta kurserna
 * och renderar ett stapeldiagram med Chart.js.
 * @async
 * @function displayStatistics
 * @returns {Promise<void>}
 */

async function displayStatistics() {
  try {
    const response = await fetch(
      "https://mallarmiun.github.io/Frontend-baserad-webbutveckling/Moment%205%20-%20Dynamiska%20webbplatser/statistik_sokande_ht25.json",
    );

    if (!response.ok) {
      throw new Error("Kunde inte hämta data från källan.");
    }

    const data = await response.json();

    const topCourses = data
      .filter((item) => item.type === "Kurs")
      .sort((a, b) => parseInt(b.applicantsTotal) - parseInt(a.applicantsTotal))
      .slice(0, 6);

    const ctx = document.getElementById("barChart");

    if (ctx) {
      const myBarChart = new Chart(ctx, {
        type: "bar",
        data: {
          labels: topCourses.map((c) => c.name),
          datasets: [
            {
              label: "Antal sökande (Totalt)",
              data: topCourses.map((c) => parseInt(c.applicantsTotal)),
              backgroundColor: "rgba(75, 192, 192, 0.5)",
              borderColor: "rgba(75, 192, 192, 1)",
              borderWidth: 1,
            },
          ],
        },
        options: {
          indexAxis: "y",
          responsive: true,
          maintainAspectRatio: false,
          color: "black",
          scales: {
            y: {
              afterFit: (axis) => {
                axis.width = 380;
              },
              ticks: {
                padding: 10,
                color: "black",
              },
            },
          },
          layout: {
            padding: {
              left: 10,
              bottom: 30,
            },
          },
          plugins: {
            legend: { position: "top" },
          },
        },
      });
    }
  } catch (error) {
    console.error("Fel:", error);
  }
}

/**
 * Initialiserar Google Maps via Google Maps Library.
 * Sätter upp kartans startposition och event listener för sökning.
 * @async
 * @function initMap
 * @returns {Promise<void>}
 */
async function initMap() {
  const [{ Map }, { AdvancedMarkerElement }] = await Promise.all([
    google.maps.importLibrary("maps"),
    google.maps.importLibrary("marker"),
  ]);

  mapElement = document.querySelector("gmp-map");
  const innerMap = mapElement.innerMap;

  marker = new AdvancedMarkerElement({
    map: innerMap,
    position: mapElement.center,
    title: "Vald plats",
  });

  document.getElementById("search-button").addEventListener("click", () => {
    const address = document.getElementById("address-input").value;
    geocodeAddress(address);
  });
}

/**
 * Utför ett AJAX-anrop (Fetch) till Google Geocoding API för att omvandla en textadress
 * till geografiska koordinater och uppdaterar kartans position.
 * @async
 * @function geocodeAddress
 * @param {string} address - Adressen som ska sökas efter.
 * @returns {Promise<void>}
 */
async function geocodeAddress(address) {
  const apiKey = "AIzaSyCea6n934RqpnE0rYs9Npf5a7llgOzhoWw";
  const url = `https://maps.googleapis.com/maps/api/geocode/json?address=${encodeURIComponent(address)}&key=${apiKey}`;

  try {
    const response = await fetch(url);
    const data = await response.json();

    if (data.status === "OK") {
      const location = data.results[0].geometry.location;

      mapElement.center = location;
      mapElement.zoom = 12;

      marker.position = location;
    } else {
      alert("Kunde inte hitta platsen: " + data.status);
    }
  } catch (error) {
    console.error("Fel vid AJAX-anrop:", error);
  }
}
